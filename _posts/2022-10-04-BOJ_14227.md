#include <iostream>
#include <cmath>
using namespace std;

int main() {
	ios::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);
	int a, b, c, d, cnt = 0; cin >> a >> b >> c >> d;
	while (1) {
		if (a == c && b == d) return !(cout << cnt);
		if (a > c || b > d) return !(cout << -1);
		if (c % 2 != d % 2) {
			if (c - a == d - b) return !(cout << c - a + cnt);
			else return !(cout << -1);
		}
		if (c % 2) {
			c--; d--; cnt++;
		}
		else {
			if (a <= c / 2 && b <= d / 2 && abs(b - a) <= abs(d / 2 - c / 2)) {
				c /= 2; d /= 2; cnt++;
			}
			else {
				if (c - a == d - b) return !(cout << c - a + cnt);
				else return !(cout << -1);
			}
		}
	}
	return 0;
}
