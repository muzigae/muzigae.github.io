#include <iostream>
#include <queue>
using namespace std;
using vi = vector<int>;

vector<vi> v(10, vi(10)), check(10, vi(10));
void sw(int a, int b) {
	check[a][b] = 1 - check[a][b];
	if (b >= 1) check[a][b - 1] = 1 - check[a][b - 1];
	if (a < 9) check[a + 1][b] = 1 - check[a + 1][b];
	if (b < 9) check[a][b + 1] = 1 - check[a][b + 1];
}

int main() {
	ios::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);
	for (int i = 0; i < 10; i++) {
		string s; cin >> s;
		for (int j = 0; j < 10; j++) v[i][j] = (s[j] == 'O');
	}

	int cnt = 200;
	for (int i = 0; i < 1024; i++) {
		check = v;
		int tmp = 0, x = 1;
		for (int j = 0; j < 10; j++) {
			if (i & (1 << j)) {
				tmp++;
				sw(0, j);
			}
		}
		for (int a = 1; a < 10; a++) {
			for (int b = 0; b < 10; b++) {
				if (check[a - 1][b]) {
					tmp++;
					sw(a, b);
				}
			}
		}
		for (int a = 0; a < 10; a++) {
			if (check[9][a]) x = 0;
		} if (x) cnt = min(cnt, tmp);
	} if (cnt != 200) cout << cnt;
	else cout << -1;
	return 0;
}
