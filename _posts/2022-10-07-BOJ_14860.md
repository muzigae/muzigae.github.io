#include <iostream>
using namespace std;
using ll = long long;

int v[15000001];
int main() {
	ios::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);
	ll a, b, mod = 1e9 + 7, res = 1; cin >> a >> b;
	for (int i = 2; i <= a; i++) {
		if (v[i]) continue;
		v[i] = 1;
		for (int j = i + i; j <= a; j += i) v[j] = 1;
		ll c = i, n = 0;
		for (ll j = i; j <= a; j *= i) n += (a / j) * (b / j);
		while (n) {
			if (n % 2) res = res * c % mod;
			c = c * c % mod;
			n /= 2;
		}
	} cout << res;
	return 0;
}
