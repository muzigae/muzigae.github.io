{% highlight C++ linenos %}
#include <iostream>
using namespace std;

int main() {
	ios::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);
	ll n, m, a; cin >> n >> m;
	for (int i = m; i >= 1; i--) {
		cin >> a;
		if (n % 2) cout << a * n << " ";
		else cout << a * (n + (1ll << i) - 2) << " ";
		n = (n + n % 2) / 2;
	}
	return 0;
}
