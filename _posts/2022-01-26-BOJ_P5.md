BOJ 1353 합과 곱

{% highlight C++ linenos %}
#include <iostream>
#include <cmath>
using namespace std;

int s, p;
double f(int x) {
	return pow((double)s / x, x);
}

int main() {
	ios::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL);
	cin >> s >> p;
	if (s == p) {
		cout << 1; return 0;
	}
	int n = 2;
	while (1) {
		if (f(n) >= p) {
			cout << n; break;
		}
		else if (f(n) > f(n + 1)) {
			cout << -1; break;
		} n++;
	}
	return 0;
}
